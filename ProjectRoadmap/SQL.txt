SELECT 
    n.nspname AS schema,
    p.proname AS function_name,
    pg_get_function_identity_arguments(p.oid) AS arguments,
    pg_get_functiondef(p.oid) AS definition
FROM pg_proc p
JOIN pg_namespace n ON p.pronamespace = n.oid
WHERE n.nspname = 'public'
ORDER BY function_name;

lấy RPC theo tên
select
    pg_get_functiondef(p.oid)
from pg_proc p
join pg_namespace n on n.oid = p.pronamespace
where n.nspname = 'public'
and p.proname = 'generate_exam';

Lấy toàn bộ index + definition (khuyến nghị)

select
    schemaname,
    tablename,
    indexname,
    indexdef
from pg_indexes
order by schemaname, tablename, indexname;

************************************************
All RPC question

select
    proname,
    pg_get_functiondef(p.oid)
from pg_proc p
join pg_namespace n on n.oid = p.pronamespace
where n.nspname = 'public'
and proname ilike '%question%';

SELECT 
    table_name, 
    column_name, 
    data_type, 
    is_nullable,
    column_default
FROM 
    information_schema.columns
WHERE 
    table_schema = 'public'
ORDER BY 
    table_name, 
    ordinal_position;

RLS all
SELECT 
    policyname AS policy_name,
    tablename AS table_name,
    schemaname AS schema_name,
    permissive,
    roles,
    cmd AS operation, -- SELECT, INSERT, UPDATE, DELETE
    qual AS check_expression, -- Chính sách cho các dòng hiện có (USING)
    with_check AS with_check_expression -- Chính sách cho các dòng mới (WITH CHECK)
FROM 
    pg_policies
WHERE 
    schemaname = 'public' 
    AND tablename = 'questions' -- Thay 'ten_bang_cua_ban' bằng tên bảng thực tế
ORDER BY 
    tablename;

5. Performance Layer (RẤT QUAN TRỌNG)
Index Strategy
Full text index
CREATE INDEX idx_questions_fts
ON questions
USING GIN (to_tsvector('english', question_text));
Filter indexes
CREATE INDEX idx_questions_type
ON questions(question_type_id);

CREATE INDEX idx_questions_category
ON questions(category_id);

CREATE INDEX idx_questions_difficulty
ON questions(difficulty);

CREATE INDEX idx_questions_active
ON questions(is_active);
Tag join index
CREATE INDEX idx_question_tags_question
ON question_tags(question_id);

CREATE INDEX idx_question_tags_tag
ON question_tags(tag_id);
6. API Contract (search_questions)

Production RPC thường nhận dạng như sau:

search_questions(

p_text_query text DEFAULT NULL,

p_category_id bigint DEFAULT NULL,
p_difficulty integer DEFAULT NULL,
p_tag_ids bigint[] DEFAULT NULL,
p_has_passage boolean DEFAULT NULL,
